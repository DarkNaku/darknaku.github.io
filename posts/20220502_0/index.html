public float GetAngle (Vector3 a, Vector3 b) {
    var dot = (a.x * b.x) + (a.y * b.y);
    var determinant = (a.x * b.y) - (a.y * b.x);
    var degree = Mathf.Atan2(determinant, dot) * Mathf.Rad2Deg;

    return (degree < 0f) ? 180f + (180f + degree) : degree;
}
